func greet(person) {
    print("Hello, " + person + "!")
}

# Control Flow

let x = 10

if x > 5 {
    print("Greater than 5")
} else {
    print("5 or less")
}

# Loops

for i in 1..5 {
    print("Count: " + i)
}

# Data Structures

let fruits = ["apple", "banana", "cherry"]

for fruit in fruits {
    print(fruit)
}
